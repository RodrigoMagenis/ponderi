/* criação da table estudante */
CREATE TABLE STUDENT (CDSTUDENT NUMERIC(10) PRIMARY KEY NOT NULL, IDSTUDENT VARCHAR(50) NOT NULL, NMSTUDENT VARCHAR(255) NULL, DTBIRTH DATETIME NULL, NMMAIL VARCHAR(255) NULL, FGGENDER NUMERIC(2) NULL);
CREATE INDEX STUDENT_CDSTUDENT ON STUDENT (CDSTUDENT);

/* Criação da table classroom */
CREATE TABLE CLASSROOM (CDCLASSROOM NUMERIC(10) PRIMARY KEY NOT NULL, IDCLASSROOM VARCHAR(50) NOT NULL, NMCLASSROOM VARCHAR(255) NULL);
CREATE INDEX CLASSROOM_CDCLASSROOM ON CLASSROOM (CDCLASSROOM);
